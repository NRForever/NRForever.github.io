function updateTimer() { const now = new Date(); const eventDate = new Date('2026-03-01T01:17:41Z'); const totalSeconds = Math.ceil((eventDate - now) / 1000); let years = Math.floor(totalSeconds / (3600 * 24 * 365)); totalSeconds %= (3600 * 24 * 365); let months = Math.floor(totalSeconds / (3600 * 24 * 30)); totalSeconds %= (3600 * 24 * 30); let days = Math.floor(totalSeconds / (3600 * 24)); totalSeconds %= (3600 * 24); let hours = Math.floor(totalSeconds / 3600); totalSeconds %= 3600; let minutes = Math.floor(totalSeconds / 60); let seconds = totalSeconds % 60; document.getElementById('timer').innerHTML = `Time left: ${years} years, ${months} months, ${days} days, ${hours} hours, ${minutes} minutes, and ${seconds} seconds`; } setInterval(updateTimer, 1000);
